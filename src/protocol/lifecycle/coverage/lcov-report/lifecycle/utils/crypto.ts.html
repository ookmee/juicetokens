import * as crypto from 'crypto';
import * as nacl from 'tweetnacl';
import * as naclUtil from 'tweetnacl-util';
 
/**
 * SHA256 hash stacking utility functions
 */
export class HashUtil {
  /**
   * Creates a SHA256 hash of the input data
   * @param data Data to hash
   * @returns SHA256 hash as Uint8Array
   */
  static sha256(data: Buffer | Uint8Array | string): Uint8Array {
    const hash = crypto.createHash('sha256');
    
    if (typeof data === 'string') {
      hash.update(data);
    } else {
      hash.update(Buffer.from(data));
    }
    
    return new Uint8Array(hash.digest());
  }
  
  /**
   * Creates a commitment using SHA256 hash stacking
   * @param data Array of data items to include in the commitment
   * @returns Commitment as Uint8Array
   */
  static createCommitment(...data: (string | Buffer | Uint8Array)[]): Uint8Array {
    // Create a composite buffer by concatenating all data elements
    const concatenated = data.map(item => {
      if (typeof item === 'string') {
        return Buffer.from(item);
      }
      return Buffer.from(item);
    }).reduce((acc, val) => Buffer.concat([acc, val]), Buffer.alloc(0));
    
    // Return the SHA256 hash of the concatenated data
    return HashUtil.sha256(concatenated);
  }
}
 
/**
 * ED25519 signature utility functions
 */
export class SignatureUtil {
  /**
   * Generates an ED25519 key pair
   * @returns Object containing publicKey and secretKey as Uint8Array
   */
  static generateKeyPair(): { publicKey: Uint8Array; secretKey: Uint8Array } {
    const keyPair = nacl.sign.keyPair();
    return {
      publicKey: keyPair.publicKey,
      secretKey: keyPair.secretKey
    };
  }
  
  /**
   * Signs data using ED25519
   * @param data Data to sign
   * @param secretKey Secret key as Uint8Array
   * @returns Signature as Uint8Array
   */
  static sign(data: string | Buffer | Uint8Array, secretKey: Uint8Array): Uint8Array {
    const message = typeof data === 'string' 
      ? naclUtil.decodeUTF8(data)
      : Buffer.from(data);
    
    return nacl.sign.detached(message, secretKey);
  }
  
  /**
   * Verifies an ED25519 signature
   * @param data Original data that was signed
   * @param signature Signature to verify
   * @param publicKey Public key as Uint8Array
   * @returns Boolean indicating if signature is valid
   */
  static verify(
    data: string | Buffer | Uint8Array,
    signature: Uint8Array,
    publicKey: Uint8Array
  ): boolean {
    const message = typeof data === 'string' 
      ? naclUtil.decodeUTF8(data)
      : Buffer.from(data);
    
    return nacl.sign.detached.verify(message, signature, publicKey);
  }
  
  /**
   * For test purposes only: Generates a deterministic signature
   * This is only for testing to ensure consistent results
   * @param data Data to sign
   * @returns Signature as Uint8Array
   */
  static deterministicSign(data: string | Buffer | Uint8Array): Uint8Array {
    // Generate a deterministic key from the data itself
    const seed = HashUtil.sha256(typeof data === 'string' ? data : Buffer.from(data)).slice(0, 32);
    const keyPair = nacl.sign.keyPair.fromSeed(seed);
    
    const message = typeof data === 'string' 
      ? naclUtil.decodeUTF8(data)
      : Buffer.from(data);
    
    return nacl.sign.detached(message, keyPair.secretKey);
  }
}
